\begin{itemize}
\tightlist
\item
  A \texttt{pdf} document uses
  \href{https://www.latex-project.org/}{latex} under the hood to compile
  a \texttt{.tex} file to \texttt{pdf}.
\item
  In latex a table is written as (see source Rmd file)
\end{itemize}

\begin{center}
\begin{tabular}{rr}
\hline
Var1 & Var2 \\
\hline
1 & 2 \\
3 & 4\\
\hline
\end{tabular}
\end{center}

\begin{itemize}
\tightlist
\item
  A markdown table is written as (see source Rmd file)
\end{itemize}

\begin{longtable}[]{@{}rr@{}}
\toprule
Var1 & Var2\tabularnewline
\midrule
\endhead
1 & 2\tabularnewline
3 & 4\tabularnewline
\bottomrule
\end{longtable}

\begin{itemize}
\tightlist
\item
  A table can be created by using \texttt{knitr}'s \texttt{kable}
  function as below:
\end{itemize}

\begin{Shaded}
\begin{Highlighting}[]
\NormalTok{df <-}\StringTok{ }\KeywordTok{data.frame}\NormalTok{(}\DataTypeTok{Var1 =} \KeywordTok{c}\NormalTok{(}\DecValTok{1}\NormalTok{, }\DecValTok{3}\NormalTok{),}
                 \DataTypeTok{Var2 =} \KeywordTok{c}\NormalTok{(}\DecValTok{2}\NormalTok{, }\DecValTok{4}\NormalTok{))}
\NormalTok{knitr}\OperatorTok{::}\KeywordTok{kable}\NormalTok{(df)}
\end{Highlighting}
\end{Shaded}

\endhead
